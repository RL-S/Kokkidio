90 /* cannot be called in device code */
91 void resize(Index rows, Index cols){
92 this->adjustDims(rows, cols);
93 if ( rows == this->rows() && cols == this->cols() ){
94 return;
95 }
96 if constexpr (
97 !std::is_const_v<EigenType_host> &&
98 !is_eigen_map_v<std::remove_const_t<EigenType_host>>
99 ){
100 /* If the ViewMap was given a (non-const) object on construction,
101 * then ViewMap::resize should be the correct way to resize both,
102 * because the Eigen object and Kokkos::View don't know about each
103 * other - i.e. there is no other non-manual way.
104 */
105 if (m_obj){
106 EigenType_host& hostObj { *(this->m_obj) };
107 /* Resize the host object */
108 hostObj.resize(rows, cols);
109 /* and if the target is the host, then the View is unmanaged,
110 * so we just wrap it around the host object */
111 if constexpr ( target == Target::host ){
112 this->wrapView(hostObj);
113 /* On the device, the View is always managed, so we resize it */
114 } else {
115 this->resizeView(rows, cols);
116 }
117 /* Without a host object, it's also always a managed View */
118 } else {
119 this->resizeView(rows, cols);
120 }
121 } else {
122 static_assert(dependent_false<EigenType_host>::value && false,
123 "Cannot resize a const or non-owning object!"
124 );
125 }

139 constexpr bool isManaged() const {
140 /* The View is only unmanaged in one case:
141 * if a host object was provided during construction AND
142 * the target (~= ExecutionSpace) is also the host.

232 if constexpr ( target == Target::host ){
233 /* If the target is the host, then the memory is already accessible,
234 * Therefore, we don't need to allocate device memory,
235 * and can simply wrap the host object's data in an unmanaged view.
236 * */
237 this->m_obj = make_observer( &hostObj );
238 this->wrapView(hostObj);
239 } else {
