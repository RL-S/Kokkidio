107template<typename _ColType>
108class HostBuffer {
109public:
110 using ColType = _ColType;
111 using Scalar = typename ColType::Scalar;
112 static constexpr auto target {Target::host};
113 using MemorySpace = Kokkidio::MemorySpace<target>;
114 using LoopType = chunk::LoopType<target, ColType>;
115
116 using DataType = Scalar***;
117 using ViewType = Kokkos::View<DataType, Kokkos::LayoutLeft, MemorySpace>;
118
119 static constexpr Index RowsAtCompileTime {ColType::RowsAtCompileTime};
120 static_assert(RowsAtCompileTime != Eigen::Dynamic);
121
122protected:
123 ViewType m_view;
124
125 template<typename Policy>
126 void set(const Policy& pol, Index chunkSizeMax){
127 auto rng { toIndexRange(pol) };
128 if constexpr (is_RangePolicy_v<Policy>){
129 chunkSizeMax = pol.chunk_size();
130 }
131 std::size_t
132 rows {RowsAtCompileTime},
133 cols {static_cast<std::size_t>( std::min(rng.size(), chunkSizeMax) )};
134 this->m_view = ViewType{
135 Kokkos::view_alloc(MemorySpace{}, Kokkos::WithoutInitializing, ""),
136 rows,
137 cols,
138 this->maxThreads()
139 };
140 printd(
141 "HostBuffer::set: range [%i, %i), chunkSizeMax = %i\n"
142 "\t(%p) View extents: (%lu, %lu, %lu)\n"
143 , rng.begin(), rng.end(), chunkSizeMax
144 , (void*) ( this->m_view.data() )
145 , this->m_view.extent(0)
146 , this->m_view.extent(1)
147 , this->m_view.extent(2)
148 );
149 }
150
151
152public:
153 static std::size_t maxThreads(){
154 #ifdef _OPENMP
155 return omp_get_max_threads();
156 #else
157 return 1;
158 #endif
159 }
160
161 HostBuffer() = default;

175 template<typename Policy>
176 HostBuffer(const Policy& pol, Index chunkSizeMax){

191 template<typename Policy>
192 HostBuffer(const Policy& pol){
193 if constexpr (is_RangePolicy_v<Policy>){
194 set( pol, pol.chunk_size() );
195 } else
196 {
197 set( pol, chunk::defaultSize );
198 }
199 }

201public:
202 LoopType get( const Chunk<Target::host>& chunk ) const {
203 #ifdef _OPENMP
204 assert( ( omp_get_max_threads() == 1 || omp_in_parallel() ) );
205 int threadNo { omp_get_thread_num() };
206 #else
207 int threadNo {0};
208 #endif
209
210 // #ifndef __CUDACC__
211 printdl(
212 "(%p) HostBuffer::get: Thread #%i, mapping to view element...\n"
213 "\tView address range: %p - %p\n"
214 "\tView extents: (%lu, %lu, %lu)\n"
215 "\tMap size: (%i, %i)\n"
216 , (void*) &( this->m_view(0, 0, threadNo) )
217 , threadNo
218 , (void*) ( this->m_view.data() )
219 , (void*) ( this->m_view.data() + this->m_view.size() )
220 , this->m_view.extent(0)
221 , this->m_view.extent(1)
222 , this->m_view.extent(2)
223 , RowsAtCompileTime, chunk.get().size()
224 );
225 // #endif
226
227 return {
228 &( this->m_view(0, 0, threadNo) ),
229 RowsAtCompileTime,
230 chunk.get().size()
231 };
232 }
